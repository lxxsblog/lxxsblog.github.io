归并大法

算法 noip  •    -  •  阅读时间: 2.0分
lxxs'blog  2020-02-08

归并绝对是需要学会手写并掌握的，他不是sort可以代替的，很多题都需要他的思想。

首先附上一般写法，这种写法是去清北学堂老师教的，不过可移植性不好。

#include<bits/stdc++.h>
using namespace std;
int q[100010],a[100010],n;
void merge_sort(int l,int r){
    if(l>=r)return;
    int mid=(l+r)>>1;
    merge_sort(l,mid);merge_sort(mid+1,r);
    int i,j,cnt = l;
    for(i = l,j = mid+1;i<=mid;i++){
        for(;q[j]<q[i]&&j<=r;j++)a[cnt++]=q[j];
        a[cnt++]=q[i];
    }
    for(;j<=r;j++)a[cnt++]=q[j];
    for(i=l;i<=r;i++)q[i]=a[i];
}

int main(){
    cin>>n;
    for(int i = 1;i<=n;i++)cin>>q[i];
    merge_sort(1,n);
    for(int i = 1;i<=n;i++)cout<<q[i]<<" ";
}

继续介绍一种更优秀的，看起来容易往别的题目移植的写法。

#include<bits/stdc++.h>
using namespace std;
int q[100010],a[100010],n;
void merge_sort(int l,int r){
    if(l>=r)return;
    int mid=(l+r)>>1;
    merge_sort(l,mid);merge_sort(mid+1,r);
    int i=l,j=mid+1,cnt = l;
    while(i<=mid&&j<=r){
        if(q[i]<=q[j])a[cnt++]=q[j];
        else a[cnt++]=q[i];
    }
    while(i<=mid)a[cnt++]=q[i];
    while(j<=r)a[cnt++]=q[j];
    for(i=l;i<=r;i++)q[i]=a[i];
}
int main(){
    cin>>n;
    for(int i = 1;i<=n;i++)cin>>q[i];
    merge_sort(1,n);
    for(int i = 1;i<=n;i++)cout<<q[i]<<" ";
}
